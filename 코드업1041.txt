import java.util.Scanner;

public class Main {
	
	public static void main(String[] args) {
		
		Scanner sc = new Scanner(System.in);
		
		
		String str = sc.next();
		
		
		char a = str.charAt(0);
		
		char result = (char)(a+1);
		
		System.out.printf("%c", result);
		
		
		
		sc.close();
		
	}
}
